function sumarMatrices(matriz1, matriz2) {
    const filas = matriz1.length;
    const columnas = matriz1[0].length;

    let matriz_result = [];

    for (let i = 0; i < filas; i++) {
        let fila_resultado = [];
        for (let j = 0; j < columnas; j++) {
            fila_resultado.push(matriz1[i][j] + matriz2[i][j]);

        matriz_result.push(fila_resultado);
    }

    console.log("Matriz resultado de la suma:");
    for (let fila of matriz_result) {
        console.log(fila);
    }
}

function sumarVectores(vector1, vector2) {
    if (vector1.length !== vector2.length) {
        throw new Error("Los vectores deben tener la misma longitud");
    }

    let vectorResultado = vector1.map((valor, index) => valor + vector2[index]);

    return vectorResultado;
}
        
def sumar_vectores(v1, v2):
    if len(v1) != len(v2):
        raise ValueError("Los vectores deben tener la misma longitud")

    return [v1[i] + v2[i] for i in range(len(v1))]
